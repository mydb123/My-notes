# 数组的遍历for/forEach()/map()

  1. for

  2. forEach（）遍历对象类型数组

// arr.forEach( function(item){
//             console.log(item)
//         } )
const arr = [
    {id:1,name:'zhangsan'},
    {id:2,name:'lisi'},
    {id:3,name:'wangwu'}
]

arr.forEach( function(item){
            console.log(item.id + '---' +  item.name)
})
// 输出结果：
// 1-----zhangsan
// 2-----lisi
// 3-----wangwu
  1. map()方法
