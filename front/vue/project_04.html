# 路由导航守卫

  • 在home.vue中判断token很麻烦
    
    router.beforeEach((to,from,next)=>{
        if(to.path === "/login"){
            next()
        }else{
            const token = localStorage.getItem('token');
            if(!token){
                Message.warning("请先登录")
                router.push({
                    name:"/login"
                })
                return
            }
            next();
        }
    })
