# Composition API

  • Composition API为vue应用提供更好的逻辑复用和代码组织
  • 之前所有写在data里面的数据都可以写在setup里
    <template>
        <div>
            <p>{{data.counter}}</p>  
        </div>
    </template>
    <script>
    import {reactive} from 'vue'
    export default {
        name:"hello word",
        props:{
            msg: String
        },
        // setup 函数是组合api的入口函数
        setup(){
            const data = reactive({
                counter:2
            })
            return {data}
        }
    }
    </script>
